@Coto: current ARM9 DTCM setup : 
__sp_undef	=	__dtcm_top - 0x200 	- 0xc;      @ONLY used on undefined exceptions
__sp_dabt	=	__sp_undef - 0x1000 - 0xc;      @ONLY used on inter_data data abort exceptions (software emulation fallback layer)
__sp_svc	=	__sp_dabt - 0x400 	- 0xc;	    @ONLY used for GBA SWI bios / NDS Hardware bios exceptions (shared between them, compatible setup so far)
__sp_swi	=	__sp_svc - 0x400 	- 0xc;	    @unused stack.
__sp_irq	=	__sp_swi  - 0x1000 	- 0xc; 	    @NDS Mode IRQ stack for emulator's IRQ handlers. GBA stack is not used here.
__sp_usr	=	__sp_irq - 0x100	- 0xc;      @unused stack.
